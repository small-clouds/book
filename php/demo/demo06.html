冒泡排序

function bubbleSort(&$myarr){
     $temp=0;
     $flag=false;
     for($i=0;$i< count($myarr)-1;$i++){
         for($j=0;$j< count($myarr)-1-$i;$j++){
             if($myarr[$j]>$myarr[$j+1]){
                   $temp=$myarr[$j];
                $myarr[$j]=$myarr[$j+1];
                $myarr[$j+1]=$temp;
                $flag = true;
             }
         }
         if(!$flag){
             break;
         }
         $flag=false;
     }
 }

选择排序

function selectSort(&$arr){
    $temp=0;
    for($i=0;$i< count($arr)-1 ;$i++){
      $minVal=$arr[$i];
      $minIndex =$i;
      for($j=$i+1;$j< count($arr);$j++){
          if($minVal>$arr[$j]){
              $minVal=$arr[$j];
              $minIndex =$j;
          }
      }
      $temp = $arr[$i];
      $arr[$i]=$arr[$minIndex];
      $arr[$minIndex]=$temp;
    }
}

插入排序

function insertSort(&$arr){
    //默认$arr[0]是有序的
    for($i=1;$i< count($arr);$i++){
        $inserVal =$arr[$i]; //带插入
        $inserIndex =$i-1; //待插入位置（和这个位置比较）
        //还没有找到时代那个的位置
            //前面还有位置&&比前面那个是小的  所有还需要和前面比
        while($inserIndex>=0&&$inserVal<$arr[$inserIndex]){
            //最小值了&&  比前面那个小
            $arr[$inserIndex+1]=$arr[$inserIndex];  //王后面挪一位
            $inserIndex--; //继续合前面位置比较
        }
        $arr[$inserIndex+1]=$inserVal; //插入 如果适当位置就不用了  
    }
}

function binarySearch(&$arr,$findVal,$leftIndex,$rightIndex){
    if($leftIndex >$rightIndex){
        echo "找不到这个数";
        return;
    }
    $middleIndex =round(($leftIndex+$rightIndex)/2);
    if($findVal > $arr[$middleIndex]){
        binarySearch($arr,$findVal,$middleIndex+1,$rightIndex);
    }else if($findVal < $arr[$middleIndex]){
        binarySearch($arr,$findVal,$leftIndex,$middleIndex-1);
    }else{
        echo "找到了该数字 下标为$middleIndex";
    }
}
