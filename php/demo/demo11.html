/ 错误日志相关
date_default_timezone_set("PRC"); // 设置中国时区
function myError($errNo, $errMsg) {
$err_info = "错误号$errNo+错误信息$errMsg";
error_log("【" . date("Y-m-d G:i:s") . "】" . $err_info . "\r\n", 3, "myError.txt"); //属于自定义错误函数的方法
}
// 设置处理函数
set_error_handler("myError", E_WARNING);
